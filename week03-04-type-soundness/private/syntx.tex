% \RRedex[n]: one line
% \RRedex[b]: two lines wrapped inside an array
% \RRedex[a]: two lines without an enclosing array
\newcommand{\NewlineRRedex}{\\}
\newcommand{\RRedex}[3][n]{
    \ifx n#1
        #2 \LArrRedex #3
    \else\ifx b#1
        \begin{array}{@{}l@{}}
            #2 \LArrRedex \NewlineRRedex
            #3
        \end{array}
    \else\ifx a#1
        #2 \LArrRedex \NewlineRRedex
        #3
    \else
        \mathrm{RRedex: Illegal~option~#1}
    \fi\fi\fi
}
